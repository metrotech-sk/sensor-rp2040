112  uint8_t offsetH = msg.at(5);
113  // convert to uint16_t
114  uint16_t offset = (offsetH << 8) + offsetL;
115 
116  // check if offset is valid (not read only memory and not negative)
117  if(offset >= READ_ONLY_OFFSET || offset < 0)
118  {
119  // send ACK_NOK

121  return;
122  }
